public class sortArray
{
public static void main(String args[])
{
Scanner sc=new Scanner(System.in);
int a[]=new int[10];
for(int i=0;i<a.length;i++)
{
System.out.println("Enter Array");
a[i]=sc.nextInt();
}
}

for(int i=0;i<a.length;i++)
{
for(int j=i+1;j<a.length;j++)
{
if(a[i]>a[j])
{
int temp=a[i];
a[i]=a[j];
a[j]=temp;
}
}
}

for(int i=0;i<a.length;i++)
{
System.out.println(a[i]);
}
}
}
